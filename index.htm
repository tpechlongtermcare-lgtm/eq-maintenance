        <div style="background:var(--primary-light);padding:10px 14px;border-radius:8px;margin-bottom:16px;font-size:13px"><i class="fas fa-info-circle" style="color:var(--primary)"></i> 設備所屬: <b>{{SD.equipment.unitName}}</b> ({{SD.equipment.unitCode}})</div>

        <div class="form-group"><label><i class="fas fa-key"></i> 原單位驗證碼（{{SD.equipment.unitName}}）</label><input v-model="LC.ownerCode" type="password" placeholder="輸入原單位驗證碼授權放行"></div>

        <div class="form-group"><label><i class="fas fa-building"></i> 借入單位</label>

          <select v-model="LC.borrowUnit" @change="LC.borrowName=allUnits.find(u=>u['單位代碼']===LC.borrowUnit)?.['單位名稱']||''">

            <option value="">-- 選擇借入單位 --</option>

            <option v-for="u in allUnits.filter(x=>String(x['單位代碼'])!==SD.equipment.unitCode&&x['啟用狀態']!=='停用')" :key="u['單位代碼']" :value="u['單位代碼']">{{u['單位名稱']}} ({{u['單位代碼']}})</option>

          </select></div>

        <div v-if="LC.borrowUnit" class="form-group"><label><i class="fas fa-key"></i> 借入單位驗證碼（{{LC.borrowName}}）</label><input v-model="LC.borrowCode" type="password" placeholder="輸入借入單位驗證碼確認接收"></div>

        <div class="form-group"><label><i class="fas fa-id-badge"></i> 經辦人員編</label>

          <div style="display:flex;gap:8px"><input v-model="LC.empId" placeholder="輸入員編" style="flex:1" @blur="lookupEmp(LC)"><div v-if="LC.empName" style="padding:10px;background:var(--primary-light);border-radius:8px;font-weight:600;white-space:nowrap">{{LC.empName}}</div></div></div>

        <button class="btn btn-warning btn-block" @click="doLoan" :disabled="!LC.ownerCode||!LC.borrowUnit||!LC.borrowCode||!LC.empId||LD"><span v-if="LD" class="loading-spinner"></span><span v-else><i class="fas fa-share"></i> 確認借出</span></button>

      </div>

 

      <!-- 歸還 -->

      <div v-if="view==='return'" class="card fade-in">

        <div class="card-header"><h3><i class="fas fa-rotate-left"></i> 儀器歸還</h3><button class="btn btn-outline btn-sm" @click="view='card'"><i class="fas fa-arrow-left"></i> 返回</button></div>

        <div style="background:#e3f2fd;padding:10px 14px;border-radius:8px;margin-bottom:16px;font-size:13px"><i class="fas fa-info-circle" style="color:#1565c0"></i> 目前持有: <b>{{SD.equipment.loanUnit||'本單位'}}</b> → 歸還至: <b>{{SD.equipment.unitName}}</b></div>

        <div v-if="SD.incomplete.length>0" style="background:#ffebee;border:2px solid var(--danger);border-radius:8px;padding:14px;margin-bottom:16px">

          <h4 style="color:var(--danger);font-size:14px;margin-bottom:8px"><i class="fas fa-ban"></i> 尚有 {{SD.incomplete.length}} 筆未完成紀錄</h4>

          <p style="font-size:13px;color:var(--danger);margin-bottom:10px">請先補齊後再歸還</p>

          <div v-for="it in SD.incomplete" :key="it.date" class="incomplete-item" style="color:var(--danger)"><span>{{it.date}}</span>

            <button class="btn btn-danger btn-sm" @click="startBackfill(it);view='backfill'"><i class="fas fa-edit"></i> 補填</button></div>

        </div>

        <div v-if="SD.incomplete.length===0">

          <div class="form-group"><label><i class="fas fa-key"></i> 借用單位驗證碼（{{SD.equipment.loanUnit}}）</label><input v-model="RC.holderCode" type="password" placeholder="輸入借用單位驗證碼"></div>

          <div class="form-group"><label><i class="fas fa-key"></i> 原單位驗證碼（{{SD.equipment.unitName}}）</label><input v-model="RC.ownerCode" type="password" placeholder="輸入原單位驗證碼確認收回"></div>

          <div class="form-group"><label><i class="fas fa-id-badge"></i> 經辦人員編</label>

            <div style="display:flex;gap:8px"><input v-model="RC.empId" placeholder="輸入員編" style="flex:1" @blur="lookupEmp(RC)"><div v-if="RC.empName" style="padding:10px;background:var(--primary-light);border-radius:8px;font-weight:600;white-space:nowrap">{{RC.empName}}</div></div></div>

          <button class="btn btn-success btn-block" @click="doReturn" :disabled="!RC.holderCode||!RC.ownerCode||!RC.empId||LD"><span v-if="LD" class="loading-spinner"></span><span v-else><i class="fas fa-check"></i> 確認歸還</span></button>

        </div>

        <button v-if="SD.incomplete.length>0" class="btn btn-block" style="background:#ccc;color:#888;cursor:not-allowed" disabled><i class="fas fa-ban"></i> 請先補齊紀錄</button>

      <!-- 操作成功 -->

<script>

// ===== 配置區（部署時修改這裡）=====

const GAS_URL = 'https://script.google.com/macros/s/AKfycbwwM2AWEGtYV0SAOff2yHp4mqzByjC1-6d-5Rx47q0dbMUQXxffs-FKsTIR8mFOIHp1kA/exec';

// =====================================

 

const UP = new URLSearchParams(location.search);

const eqCode = UP.get('eq') || '';

 

async function api(action, data = {}) {

  const resp = await fetch(GAS_URL, {

    method: 'POST', headers: { 'Content-Type': 'text/plain' },

    body: JSON.stringify({ action, data }), redirect: 'follow'

  });

  if (!resp.ok) throw new Error('HTTP ' + resp.status);

  return resp.json();

}

 

const { createApp, ref, reactive, computed, onMounted } = Vue;

createApp({

  setup() {

    const LD = ref(false);

    const toast = reactive({ show: false, message: '', type: 'success' });

    const view = ref('card');

    const history = ref([]);

    const historyAll = ref(false);

    const allUnits = ref([]);

    const doneMsg = ref('');

    const SD = reactive({ equipment: null, incomplete: [], error: null, backfillDate: '' });

    const SF = reactive({ operator: '', operatorName: '', checks: ['','','','',''], tryFix: '', onlineRepair: '', markBroken: '', note: '' });

    const LC = reactive({ ownerCode: '', borrowUnit: '', borrowName: '', borrowCode: '', empId: '', empName: '' });

    const RC = reactive({ holderCode: '', ownerCode: '', empId: '', empName: '' });

 

    const checkItems = [

      { label: '燈號面板正常', icon: 'fas fa-lightbulb' },

      { label: '警報鳴響正常', icon: 'fas fa-bell' },

      { label: '電源線外觀無損毀', icon: 'fas fa-plug' },

      { label: '按鍵開關功能正常', icon: 'fas fa-keyboard' },

      { label: '外觀清潔無黏貼雜物', icon: 'fas fa-broom' }

    ];

    const hasAbn = computed(() => SF.checks.includes('異常'));

 

    function showToast(m, t = 'success') { toast.message = m; toast.type = t; toast.show = true; setTimeout(() => toast.show = false, 3000); }

    function dayN(d) { return ['日','一','二','三','四','五','六'][parseInt(d)] || d; }

 

    async function loadData() {

      if (!eqCode) { SD.error = '缺少儀器編碼（請掃描 QR Code）'; return; }

      try {

        const r = await api('getScanPageData', { code: eqCode });

        if (r && r.success) {

          SD.equipment = r.equipment;

          SD.incomplete = r.incomplete || [];

          history.value = r.history || [];

          allUnits.value = r.units || [];

        } else { SD.error = (r && r.error) || '載入失敗'; }

      } catch (e) { SD.error = '連線失敗: ' + (e.message || String(e)); }

    }

 

    async function reload() {

      try {

        const r = await api('getScanPageData', { code: eqCode });

        if (r && r.success) { SD.equipment = r.equipment; SD.incomplete = r.incomplete || []; history.value = r.history || []; allUnits.value = r.units || []; }

      } catch (e) {}

    }

 

    async function lookupOperator() {

      if (!SF.operator) { SF.operatorName = ''; return; }

      try { const r = await api('lookupStaffByEmpId', { empId: SF.operator }); SF.operatorName = r.success ? r.name : '查無此員編'; } catch (e) { SF.operatorName = ''; }

    }

 

    async function lookupEmp(obj) {

      if (!obj.empId) { obj.empName = ''; return; }

      try { const r = await api('lookupStaffByEmpId', { empId: obj.empId }); obj.empName = r.success ? r.name : '查無此員編'; } catch (e) { obj.empName = ''; }

    }

 

    async function submitScan() {

      if (!SF.operator) { showToast('請輸入員編', 'error'); return; }

      if (SF.checks.some(c => !c)) { showToast('請完成所有檢查項目', 'error'); return; }

      LD.value = true;

      const opStr = SF.operator + (SF.operatorName ? ' ' + SF.operatorName : '');

      try {

        const r = await api('submitMaintenance', { code: eqCode, operator: opStr, check1: SF.checks[0], check2: SF.checks[1], check3: SF.checks[2], check4: SF.checks[3], check5: SF.checks[4], tryFix: SF.tryFix, onlineRepair: SF.onlineRepair, markBroken: SF.markBroken, note: SF.note });

        if (r.success) { doneMsg.value = '保養紀錄已送出！'; view.value = 'done'; SF.checks = ['','','','','']; SF.tryFix = ''; SF.onlineRepair = ''; SF.markBroken = ''; SF.note = ''; await reload(); }

        else showToast(r.error, 'error');

      } catch (e) { showToast('操作失敗', 'error'); }

      LD.value = false;

    }

 

    function startBackfill(it) { SD.backfillDate = it.date; SF.checks = ['','','','','']; SF.tryFix = ''; SF.onlineRepair = ''; SF.markBroken = ''; SF.note = ''; }

 

    async function submitBackfill() {

      if (!SF.operator || SF.checks.some(c => !c)) { showToast('請填完所有項目', 'error'); return; }

      LD.value = true;

      try {

        const r = await api('submitBackfill', { code: eqCode, operator: SF.operator, originalDate: SD.backfillDate, check1: SF.checks[0], check2: SF.checks[1], check3: SF.checks[2], check4: SF.checks[3], check5: SF.checks[4], tryFix: SF.tryFix, onlineRepair: SF.onlineRepair, markBroken: SF.markBroken, note: SF.note });

        if (r.success) { doneMsg.value = '補填紀錄已送出！'; view.value = 'done'; await reload(); }

        else showToast(r.error, 'error');

      } catch (e) { showToast('操作失敗', 'error'); }

      LD.value = false;

    }

 

    async function doLoan() {

      LD.value = true;

      try {

        const r = await api('loanByCode', { eqCode: eqCode, ownerVerifyCode: LC.ownerCode, borrowerVerifyCode: LC.borrowCode, borrowUnitCode: LC.borrowUnit, operatorEmpId: LC.empId, operatorName: LC.empName });

        if (r.success) { doneMsg.value = r.message; view.value = 'done'; LC.ownerCode = ''; LC.borrowUnit = ''; LC.borrowCode = ''; LC.empId = ''; LC.empName = ''; await reload(); }

        else showToast(r.error, 'error');

      } catch (e) { showToast('操作失敗', 'error'); }

      LD.value = false;

    }

 

    async function doReturn() {

      LD.value = true;

      try {

        const r = await api('returnByCode', { eqCode: eqCode, holderVerifyCode: RC.holderCode, ownerVerifyCode: RC.ownerCode, operatorEmpId: RC.empId, operatorName: RC.empName });

        if (r.success) { doneMsg.value = r.message; view.value = 'done'; RC.holderCode = ''; RC.ownerCode = ''; RC.empId = ''; RC.empName = ''; await reload(); }

        else showToast(r.error, 'error');

      } catch (e) { showToast('操作失敗', 'error'); }

      LD.value = false;

    }

 

    onMounted(loadData);

 

    return { LD, toast, view, history, historyAll, allUnits, doneMsg, SD, SF, LC, RC, checkItems, hasAbn, showToast, dayN, loadData, reload, lookupOperator, lookupEmp, submitScan, startBackfill, submitBackfill, doLoan, doReturn };

  }

}).mount('#app');

</script>
